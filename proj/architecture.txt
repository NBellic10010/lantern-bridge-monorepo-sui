```move
// 用戶份額記錄
struct UserPosition has key, store {
    id: UID,
    user: address,
    shares: u64,
    deposit_timestamp: u64,
}

/// 存款：一步式存入 + 轉入 Navi
public entry fun deposit<T>(
    vault: &mut Vault<T>,
    user_pos: &mut UserPosition,
    coin: Coin<T>,
    ctx: &mut TxContext
) {
    // 1. 計算份額
    let assets = coin.value();
    let shares = if (vault.total_shares == 0) {
        assets  // 初始匯率 1:1
    } else {
        (assets * vault.total_shares) / vault.total_assets
    };
    
    // 2. 轉入 Vault
    let balance = coin.into_balance();
    vault.total_assets = vault.total_assets + assets;
    vault.total_shares = vault.total_shares + shares;
    
    // 3. 存入 Navi（調用外部協議）
    // ... Navi integration
    
    // 4. 記錄用戶份額
    user_pos.shares = user_pos.shares + shares;
}
```

```typescript
// services/gas-sponsor.service.ts
interface GasSponsorRequest {
  userAddress: string;
  unsignedPTB: string;  // Base64 encoded
  depositAmount: number;
}

class GasSponsorService {
  async sponsorTransaction(req: GasSponsorRequest): Promise<SponsoredTx> {
    // 1. 獲取用戶錢包地址
    const userAddress = req.userAddress;
    
    // 2. 請求 Shinami Gas Station
    const sponsorResponse = await this.shinamiClient.sponsorTransaction({
      sender: userAddress,
      tx: req.unsignedPTB,
      gasBudget: 1000000, // 1 SUI
    });
    
    // 3. 後端簽名（使用服務器私鑰）
    const backendSignature = await this.signTransaction(
      sponsorResponse.txBytes
    );
    
    // 4. 返回完整的 Sponsored PTB
    return {
      txBytes: sponsorResponse.txBytes,
      sponsorSignature: sponsorResponse.signature,
      backendSignature,
      gasUsed: sponsorResponse.gasUsed,
    };
  }
}
```

#### 🔹 Anti-Sybil Engine（防女巫風控）

```typescript
// services/anti-sybil.service.ts
interface SybilCheckRule {
  minDepositAmount: number;      // 最低存款金額
  maxDailySponsorPerAddress: number;  // 每日最大代付次數
  cooldownPeriodMs: number;       // 冷卻時間
}

class AntiSybilEngine {
  private rules: SybilCheckRule = {
    minDepositAmount: 50,         // $50 USD
    maxDailySponsorPerAddress: 1, // 每日 1 次
    cooldownPeriodMs: 24 * 60 * 60 * 1000, // 24 小時
  };
  
  async check(userAddress: string, depositAmount: number): Promise<SybilResult> {
    // 1. 檢查存款金額
    if (depositAmount < this.rules.minDepositAmount) {
      return { allowed: false, reason: 'MIN_DEPOSIT_NOT_MET' };
    }
    
    // 2. 檢查 Redis 緩存（高性能）
    const cacheKey = `sponsor:${userAddress}:${this.getDateKey()}`;
    const cached = await this.redis.get(cacheKey);
    
    if (cached) {
      return { allowed: false, reason: 'DAILY_LIMIT_EXCEEDED' };
    }
    
    // 3. 檢查 PostgreSQL（持久化記錄）
    const todayCount = await this.db.sponsorCount.findMany({
      where: {
        userAddress,
        createdAt: { gte: this.getTodayStart() },
      },
    });
    
    if (todayCount >= this.rules.maxDailySponsorPerAddress) {
      return { allowed: false, reason: 'DAILY_LIMIT_EXCEEDED' };
    }
    
    // 4. 記錄此次請求
    await this.db.sponsorCount.create({
      userAddress,
      depositAmount,
      ipAddress: this.getClientIP(),
      userAgent: this.getClientUA(),
    });
    
    // 5. 寫入 Redis 緩存
    await this.redis.setex(cacheKey, 86400, '1');
    
    return { allowed: true };
  }
}
```

#### 🔹 RPC Load Balancer（RPC 負載均衡）

```typescript
// services/rpc-balancer.service.ts
interface RPCNode {
  url: string;
  weight: number;
  healthy: boolean;
  latency: number;
}

class RPCLoadBalancer {
  private nodes: RPCNode[] = [
    { url: 'https://rpc.mainnet.sui.io', weight: 2, healthy: true, latency: 0 },
    { url: 'https://quicknode.rpc.sui', weight: 1, healthy: true, latency: 0 },
    { url: 'https://alchemy.rpc.sui', weight: 1, healthy: true, latency: 0 },
  ];
  
  async execute<T>(fn: (rpc: string) => Promise<T>): Promise<T> {
    const availableNodes = this.nodes.filter(n => n.healthy);
    
    for (const node of availableNodes) {
      try {
        // 測量延遲
        const start = Date.now();
        const result = await fn(node.url);
        node.latency = Date.now() - start;
        
        return result;
      } catch (error) {
        // 節點故障，標記為不健康
        if (this.isFatalError(error)) {
          node.healthy = false;
          console.warn(`RPC node ${node.url} marked unhealthy`);
        }
      }
    }
    
    throw new Error('All RPC nodes failed');
  }
}
```

### ADR-001: 採用單一合約 Package 設計

| 項目 | 內容 |
|------|------|
| **背景** | V1.0 功能相對簡單（存款、提款、簡單費率） |
| **選項** | 1. 單一 Package 多 Module<br>2. 多 Package 分散部署 |
| **決策** | ✅ 選項 1 |
| **理由** | 降低部署複雜度，減少跨 Package 調用開銷，MVP 階段更易於審計 |

### ADR-002: 後端風控使用 Redis + PostgreSQL 混合存儲

| 項目 | 內容 |
|------|------|
| **背景** | 需要高性能的風控檢查與持久化記錄 |
| **選項** | 1. 純 PostgreSQL<br>2. 純 Redis<br>3. Redis + PostgreSQL 混合 |
| **決策** | ✅ 選項 3 |
| **理由** | Redis 支撐高並發讀取（風控檢查），PostgreSQL 支撐持久化與審計需求 |

| 項目 | 內容 |
|------|------|

| **選項** | 1. Shinami Sponsored API<br>2. Mysten Enoki<br>3. 自建 Gas Station |
| **決策** | ✅ 選項 1 (Shinami) |

## 5️⃣ 關鍵設計決策

### ADR-004: 跨鏈模式選擇

| 項目 | 內容 |
|------|------|
| **背景** | 需要支持 EVM ↔ Sui 雙向跨鏈生息 |
| **選項** | 1. 托管模式 (Vault 持有資產)<br>2. 非托管模式 |
| **決策** | ✅ 選項 1 (初期) |
| **理由** | 簡單安全，用戶體驗好，適合 MVP |

### ADR-005: 收益來源選擇

| 項目 | 內容 |
|------|------|
| **背景** | 需要選擇 EVM 和 Sui 的收益來源 |
| **選項** | 1. 單一來源<br>2. 多策略分散 |
| **決策** | ✅ 選項 2 |
| **理由** | 分散風險，提高收益穩定性 |

### ADR-006: Relayer 選擇

| 項目 | 內容 |
|------|------|
| **背景** | 需要選擇跨鏈消息中繼方案 |
| **選項** | 1. Wormhole 官方 Relayer<br>2. 自建 Relayer |
| **決策** | ✅ 選項 1 (初期) |
| **理由** | 降低運維複雜度，快速上線 |
